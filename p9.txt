minikube start --driver=docker

minikube status

nano hello.yaml

// hello.yaml
apiVersion: v1
kind: Pod
metadata: 
  name: hello-pod
  labels:
    app: hello
spec:
  containers:
  - name: hello-container
    image: ngix
    ports:
    - containerPort: 80
---
apiVersion: v1
kind: Service
metadata: 
  name: hello-service
spec:
  type: NodePort
  selector:
    app: hello
  ports:
  - port: 80
    targetPort: 80
    nodePort: 30007

// press ctrl X -> y -> enter

// in terminal
kubectl apply -f hello.yaml
kubectl get pods
kubectl get svc
