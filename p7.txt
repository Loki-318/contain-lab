import os
import sys
if len(sys.argv) != 2:
  print("Usage: python ping.py <hostname>")
  sys.exit(1)
host = sys.argv[1]
res = os.system(f"ping -c 4 {host}")
if res == 0:
  print(f"{host} is rwachable")
else:
  print(f"{host} is unreachable")

FROM python:3.9-slim
RUN apt-get install -y iputils-ping && rm -rf /var/lib/apt/lists/*
COPY ping.py /ping.py
CMD ["python", "ping.py']
