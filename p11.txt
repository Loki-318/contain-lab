import os
import sys
if len(sys.argv) != 2:
  print("Usage: python ping.py <hostname>")
  sys.exit(1)
host = sys.argv[1]
res = os.system(f"ping -c 4 {host}")
if res == 0:
  print(f"{host} is reachable")
else:
  print(f"{host} is not reachable")

FROM python:3.9-slim
WORKDIR /app
COPY ping.py /ping.py
ENTRYPOINT ["python", "/ping.py"]
